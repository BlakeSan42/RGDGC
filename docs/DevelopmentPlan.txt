# River Grove Disc Golf Club (RGDGC) - Complete Development Plan

## Executive Summary

This document outlines a comprehensive disc golf application for River Grove Disc Golf Club, inspired by UDisc’s functionality but with added blockchain integration for club membership, payments, and rewards. The application will be a world-class, mobile-first Progressive Web App (PWA) designed for Claude Code to implement.

### Core Services Implementation

#### Points Calculation Service (Updated for Disc Golf)

```python
# app/services/points_service.py
"""
Points calculation for RGDGC leagues.

Scoring System:
- Points = Number of participants - finish position + 1
- Example: 8 players → 1st=8pts, 2nd=7pts, ..., 8th=1pt
- Ties: Same rank, next position skipped
- DNF/DQ: 0 points
"""

from typing import List, Dict, Optional
from dataclasses import dataclass
from enum import Enum


class FinishStatus(Enum):
    COMPLETED = "completed"
    DNF = "dnf"  # Did not finish
    DQ = "dq"    # Disqualified


@dataclass
class PlayerResult:
    player_id: int
    player_name: str
    total_score: int  # Strokes relative to par
    raw_strokes: int  # Total strokes
    status: FinishStatus = FinishStatus.COMPLETED


@dataclass
class RankedResult:
    player_id: int
    player_name: str
    total_score: int
    raw_strokes: int
    finish_position: int
    points_earned: int
    status: FinishStatus


def calculate_positions(results: List[PlayerResult]) -> List[RankedResult]:
    """
    Calculate finish positions and points from raw scores.
    
    Lower score = better (golf scoring).
    Ties receive the same position.
    """
    # Separate completed from DNF/DQ
    completed = [r for r in results if r.status == FinishStatus.COMPLETED]
    incomplete = [r for r in results if r.status != FinishStatus.COMPLETED]
    
    # Sort completed by score (ascending - lower is better in golf)
    sorted_results = sorted(completed, key=lambda x: x.total_score)
    
    num_participants = len(results)
    ranked_results = []
    
    current_position = 1
    i = 0
    
    while i < len(sorted_results):
        # Find all players tied at this score
        current_score = sorted_results[i].total_score
        tied_players = []
        
        while i < len(sorted_results) and sorted_results[i].total_score == current_score:
            tied_players.append(sorted_results[i])
            i += 1
        
        # All tied players get the same position and points
        points = calculate_points(num_participants, current_position)
        
        for player in tied_players:
            ranked_results.append(RankedResult(
                player_id=player.player_id,
                player_name=player.player_name,
                total_score=player.total_score,
                raw_strokes=player.raw_strokes,
                finish_position=current_position,
                points_earned=points,
                status=player.status
            ))
        
        # Next position accounts for ties
        current_position += len(tied_players)
    
    # DNF/DQ players get 0 points
    for player in incomplete:
        ranked_results.append(RankedResult(
            player_id=player.player_id,
            player_name=player.player_name,
            total_score=player.total_score,
            raw_strokes=player.raw_strokes,
            finish_position=len(sorted_results) + 1,  # Last position
            points_earned=0,
            status=player.status
        ))
    
    return ranked_results


def calculate_points(num_participants: int, position: int) -> int:
    """
    Calculate points for a given position.
    
    Formula: points = num_participants - position + 1
    
    Examples:
        8 players: 1st=8, 2nd=7, 3rd=6, ..., 8th=1
        12 players: 1st=12, 2nd=11, ..., 12th=1
    """
    if position > num_participants:
        return 0
    
    points = num_participants - position + 1
    return max(points, 0)


def calculate_doubles_points(
    team_results: List[Dict],
    num_teams: int
) -> List[Dict]:
    """
    Calculate points for doubles/team events.
    
    Each team member receives the same points.
    Points based on team finishing position.
    """
    ranked_teams = []
    
    # Sort teams by score
    sorted_teams = sorted(team_results, key=lambda x: x["total_score"])
    
    current_position = 1
    i = 0
    
    while i < len(sorted_teams):
        current_score = sorted_teams[i]["total_score"]
        tied_teams = []
        
        while i < len(sorted_teams) and sorted_teams[i]["total_score"] == current_score:
            tied_teams.append(sorted_teams[i])
            i += 1
        
        points = calculate_points(num_teams, current_position)
        
        for team in tied_teams:
            ranked_teams.append({
                **team,
                "finish_position": current_position,
                "points_per_player": points
            })
        
        current_position += len(tied_teams)
    
    return ranked_teams


def generate_leaderboard(
    season_results: List[Dict],
    drop_worst: int = 0
) -> List[Dict]:
    """
    Generate season leaderboard from all event results.
    
    Args:
        season_results: List of all player results for the season
        drop_worst: Number of worst rounds to drop (optional)
    
    Returns:
        Sorted leaderboard with accumulated stats
    """
    # Aggregate by player
    player_stats = {}
    
    for result in season_results:
        player_id = result["player_id"]
        
        if player_id not in player_stats:
            player_stats[player_id] = {
                "player_id": player_id,
                "player_name": result["player_name"],
                "events_played": 0,
                "total_points": 0,
                "all_points": [],
                "wins": 0,
                "podiums": 0,
                "top_5": 0,
                "best_finish": None,
                "worst_finish": None,
            }
        
        stats = player_stats[player_id]
        stats["events_played"] += 1
        stats["all_points"].append(result["points_earned"])
        
        position = result["finish_position"]
        
        if stats["best_finish"] is None or position < stats["best_finish"]:
            stats["best_finish"] = position
        if stats["worst_finish"] is None or position > stats["worst_finish"]:
            stats["worst_finish"] = position
        
        if position == 1:
            stats["wins"] += 1
        if position <= 3:
            stats["podiums"] += 1
        if position <= 5:
            stats["top_5"] += 1
    
    # Calculate totals (with optional drop worst)
    leaderboard = []
    
    for player_id, stats in player_stats.items():
        points_list = sorted(stats["all_points"], reverse=True)
        
        if drop_worst > 0 and len(points_list) > drop_worst:
            counted_points = points_list[:-drop_worst]
        else:
            counted_points = points_list
        
        stats["total_points"] = sum(counted_points)
        stats["counted_events"] = len(counted_points)
        stats["average_points"] = round(
            stats["total_points"] / max(len(counted_points), 1), 2
        )
        
        del stats["all_points"]  # Remove raw list
        leaderboard.append(stats)
    
    # Sort by total points (desc), then wins, then average
    leaderboard.sort(
        key=lambda x: (-x["total_points"], -x["wins"], -x["average_points"])
    )
    
    # Assign ranks with tie handling
    current_rank = 1
    prev_points = None
    
    for i, entry in enumerate(leaderboard):
        if prev_points is not None and entry["total_points"] == prev_points:
            entry["rank"] = leaderboard[i - 1]["rank"]
        else:
            entry["rank"] = current_rank
        
        prev_points = entry["total_points"]
        current_rank = i + 2
    
    return leaderboard
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/token/ERC20/extensions/ERC20Burnable.sol";
import "@openzeppelin/contracts/access/AccessControl.sol";
import "@openzeppelin/contracts/security/Pausable.sol";

/**
 * @title RGDGToken
 * @dev ERC-20 token for River Grove Disc Golf Club
 * 
 * Features:
 * - Fixed supply (minted at deployment)
 * - Role-based access control
 * - Pausable in emergencies
 * - Burnable by holders
 */
contract RGDGToken is ERC20, ERC20Burnable, AccessControl, Pausable {
    bytes32 public constant ADMIN_ROLE = keccak256("ADMIN_ROLE");
    bytes32 public constant TREASURER_ROLE = keccak256("TREASURER_ROLE");
    
    uint256 public constant INITIAL_SUPPLY = 1_000_000 * 10**18; // 1M tokens
    
    // Events
    event TokensPurchased(address indexed buyer, uint256 amount, uint256 ethPaid);
    event PriceUpdated(uint256 newPrice);
    
    // Token price in wei (e.g., 0.001 ETH per token)
    uint256 public tokenPrice = 0.001 ether;
    
    constructor(address treasury) ERC20("River Grove Disc Golf", "RGDG") {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(ADMIN_ROLE, msg.sender);
        _grantRole(TREASURER_ROLE, treasury);
        
        // Mint initial supply to treasury
        _mint(treasury, INITIAL_SUPPLY);
    }
    
    /**
     * @dev Purchase tokens with ETH
     */
    function buyTokens() external payable whenNotPaused {
        require(msg.value > 0, "Must send ETH");
        
        uint256 tokenAmount = (msg.value * 10**18) / tokenPrice;
        require(tokenAmount > 0, "Insufficient ETH for minimum purchase");
        
        // Transfer from treasury (requires treasury approval)
        address treasury = getRoleMember(TREASURER_ROLE, 0);
        require(balanceOf(treasury) >= tokenAmount, "Insufficient treasury balance");
        
        _transfer(treasury, msg.sender, tokenAmount);
        
        // Send ETH to treasury
        payable(treasury).transfer(msg.value);
        
        emit TokensPurchased(msg.sender, tokenAmount, msg.value);
    }
    
    /**
     * @dev Update token price (admin only)
     */
    function setTokenPrice(uint256 newPrice) external onlyRole(ADMIN_ROLE) {
        require(newPrice > 0, "Price must be positive");
        tokenPrice = newPrice;
        emit PriceUpdated(newPrice);
    }
    
    /**
     * @dev Pause all transfers (emergency)
     */
    function pause() external onlyRole(ADMIN_ROLE) {
        _pause();
    }
    
    function unpause() external onlyRole(ADMIN_ROLE) {
        _unpause();
    }
    
    /**
     * @dev Hook to check pause state
     */
    function _beforeTokenTransfer(
        address from,
        address to,
        uint256 amount
    ) internal override whenNotPaused {
        super._beforeTokenTransfer(from, to, amount);
    }
}
```

#### Treasury.sol

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/access/AccessControl.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

/**
 * @title Treasury
 * @dev Manages RGDGC club funds and league fee collection
 */
contract Treasury is AccessControl, ReentrancyGuard {
    bytes32 public constant ADMIN_ROLE = keccak256("ADMIN_ROLE");
    
    IERC20 public immutable rgdgToken;
    
    // Fee configuration
    uint256 public leagueFee = 5 * 10**18;  // 5 RGDG per event
    uint256 public membershipFee = 50 * 10**18;  // 50 RGDG annual
    
    // Tracking
    mapping(address => bool) public isActiveMember;
    mapping(address => uint256) public memberExpiry;
    mapping(bytes32 => bool) public paidEvents;  // eventId => paid
    
    // Events
    event LeagueFeePaid(address indexed player, bytes32 eventId, uint256 amount);
    event MembershipPurchased(address indexed member, uint256 expiry);
    event FeeUpdated(string feeType, uint256 newAmount);
    event FundsWithdrawn(address indexed admin, uint256 amount);
    
    constructor(address _rgdgToken) {
        rgdgToken = IERC20(_rgdgToken);
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(ADMIN_ROLE, msg.sender);
    }
    
    /**
     * @dev Pay league event fee
     */
    function payLeagueFee(bytes32 eventId) external nonReentrant {
        bytes32 paymentKey = keccak256(abi.encodePacked(msg.sender, eventId));
        require(!paidEvents[paymentKey], "Already paid for this event");
        
        require(
            rgdgToken.transferFrom(msg.sender, address(this), leagueFee),
            "Token transfer failed"
        );
        
        paidEvents[paymentKey] = true;
        emit LeagueFeePaid(msg.sender, eventId, leagueFee);
    }
    
    /**
     * @dev Check if player has paid for event
     */
    function hasPlayerPaid(address player, bytes32 eventId) external view returns (bool) {
        return paidEvents[keccak256(abi.encodePacked(player, eventId))];
    }
    
    /**
     * @dev Purchase annual membership
     */
    function purchaseMembership() external nonReentrant {
        require(
            rgdgToken.transferFrom(msg.sender, address(this), membershipFee),
            "Token transfer failed"
        );
        
        uint256 newExpiry;
        if (memberExpiry[msg.sender] > block.timestamp) {
            // Extend existing membership
            newExpiry = memberExpiry[msg.sender] + 365 days;
        } else {
            // New membership
            newExpiry = block.timestamp + 365 days;
        }
        
        memberExpiry[msg.sender] = newExpiry;
        isActiveMember[msg.sender] = true;
        
        emit MembershipPurchased(msg.sender, newExpiry);
    }
    
    /**
     * @dev Update fees (admin only)
     */
    function setLeagueFee(uint256 newFee) external onlyRole(ADMIN_ROLE) {
        leagueFee = newFee;
        emit FeeUpdated("league", newFee);
    }
    
    function setMembershipFee(uint256 newFee) external onlyRole(ADMIN_ROLE) {
        membershipFee = newFee;
        emit FeeUpdated("membership", newFee);
    }
    
    /**
     * @dev Withdraw funds (admin only)
     */
    function withdrawTokens(uint256 amount) external onlyRole(ADMIN_ROLE) nonReentrant {
        require(rgdgToken.transfer(msg.sender, amount), "Transfer failed");
        emit FundsWithdrawn(msg.sender, amount);
    }
    
    /**
     * @dev Withdraw ETH (admin only)
     */
    function withdrawETH() external onlyRole(ADMIN_ROLE) nonReentrant {
        uint256 balance = address(this).balance;
        require(balance > 0, "No ETH balance");
        payable(msg.sender).transfer(balance);
    }
    
    /**
     * @dev Check treasury balances
     */
    function getBalances() external view returns (uint256 tokenBalance, uint256 ethBalance) {
        tokenBalance = rgdgToken.balanceOf(address(this));
        ethBalance = address(this).balance;
    }
    
    // Allow receiving ETH
    receive() external payable {}
}
```

```typescript
// mcp-server/src/index.ts
import { Server } from "@modelcontextprotocol/sdk/server";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio";

const server = new Server({
  name: "rgdgc-mcp",
  version: "1.0.0",
});

// Tool: Get Leaderboard
server.setRequestHandler("tools/call", async (request) => {
  if (request.params.name === "get_leaderboard") {
    const { league_id, limit = 10 } = request.params.arguments;
    
    // Call RGDGC API
    const response = await fetch(
      `${API_BASE}/api/v1/leaderboard/${league_id}?limit=${limit}`
    );
    const data = await response.json();
    
    // Format for Claude
    let result = `📊 **${data.league_name} Standings**\n\n`;
    data.entries.forEach((entry, i) => {
      const medal = i === 0 ? "🥇" : i === 1 ? "🥈" : i === 2 ? "🥉" : `${i + 1}.`;
      result += `${medal} ${entry.player_name}: ${entry.total_points} pts (${entry.events_played} events)\n`;
    });
    
    return { content: [{ type: "text", text: result }] };
  }
  
  // ... other tools
});

// Resource: PDGA Rules
server.setRequestHandler("resources/read", async (request) => {
  if (request.params.uri === "rules://pdga") {
    const rules = await loadPDGARules();
    return { contents: [{ uri: request.params.uri, text: rules }] };
  }
});

// Start server
const transport = new StdioServerTransport();
server.connect(transport);
```

### SQL Schema (PostgreSQL)

```sql
-- Users table
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255),
    display_name VARCHAR(100),
    phone VARCHAR(20),
    avatar_url TEXT,
    role VARCHAR(20) DEFAULT 'player' CHECK (role IN ('admin', 'player', 'guest')),
    wallet_address VARCHAR(42),  -- Ethereum address (P1)
    handicap DECIMAL(4,1),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Courses table
CREATE TABLE courses (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    location VARCHAR(255),
    city VARCHAR(100),
    state VARCHAR(50),
    country VARCHAR(50) DEFAULT 'USA',
    latitude DECIMAL(10, 7),
    longitude DECIMAL(10, 7),
    description TEXT,
    photo_url TEXT,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Course layouts (multiple per course)
CREATE TABLE layouts (
    id SERIAL PRIMARY KEY,
    course_id INTEGER REFERENCES courses(id) ON DELETE CASCADE,
    name VARCHAR(50) NOT NULL,  -- "Red", "White", "Blue", "Championship"
    holes INTEGER DEFAULT 18,
    total_par INTEGER NOT NULL,
    total_distance INTEGER,  -- feet
    difficulty VARCHAR(20),  -- "beginner", "intermediate", "advanced"
    is_default BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Individual holes
CREATE TABLE holes (
    id SERIAL PRIMARY KEY,
    layout_id INTEGER REFERENCES layouts(id) ON DELETE CASCADE,
    hole_number INTEGER NOT NULL,
    par INTEGER NOT NULL CHECK (par >= 2 AND par <= 6),
    distance INTEGER,  -- feet
    description TEXT,
    photo_url TEXT,
    UNIQUE(layout_id, hole_number)
);

-- Rounds (single player sessions)
CREATE TABLE rounds (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    layout_id INTEGER REFERENCES layouts(id),
    started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    total_score INTEGER,  -- relative to par
    total_strokes INTEGER,
    weather VARCHAR(50),
    wind VARCHAR(50),
    temperature INTEGER,
    notes TEXT,
    is_practice BOOLEAN DEFAULT FALSE
);

-- Hole-by-hole scores
CREATE TABLE hole_scores (
    id SERIAL PRIMARY KEY,
    round_id INTEGER REFERENCES rounds(id) ON DELETE CASCADE,
    hole_id INTEGER REFERENCES holes(id),
    strokes INTEGER NOT NULL,
    putts INTEGER,
    fairway_hit BOOLEAN,
    green_in_regulation BOOLEAN,
    ob_strokes INTEGER DEFAULT 0,
    penalty_strokes INTEGER DEFAULT 0,
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(round_id, hole_id)
);

-- Leagues
CREATE TABLE leagues (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    season VARCHAR(20),  -- "2025", "Spring 2025"
    league_type VARCHAR(20) CHECK (league_type IN ('singles', 'doubles', 'teams')),
    points_rule VARCHAR(50) DEFAULT 'field_size',  -- how points are calculated
    drop_worst INTEGER DEFAULT 0,  -- number of worst events to drop
    max_events INTEGER,  -- cap on events counted
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- League events (individual matches/weeks)
CREATE TABLE events (
    id SERIAL PRIMARY KEY,
    league_id INTEGER REFERENCES leagues(id) ON DELETE CASCADE,
    layout_id INTEGER REFERENCES layouts(id),
    name VARCHAR(100),
    event_date DATE NOT NULL,
    status VARCHAR(20) DEFAULT 'upcoming' CHECK (status IN ('upcoming', 'active', 'completed', 'cancelled')),
    num_players INTEGER,
    entry_fee DECIMAL(10,2),
    entry_fee_rgdg INTEGER,  -- P1: token fee
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Teams (for doubles/team events)
CREATE TABLE teams (
    id SERIAL PRIMARY KEY,
    event_id INTEGER REFERENCES events(id) ON DELETE CASCADE,
    name VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Team membership
CREATE TABLE team_members (
    id SERIAL PRIMARY KEY,
    team_id INTEGER REFERENCES teams(id) ON DELETE CASCADE,
    user_id INTEGER REFERENCES users(id),
    UNIQUE(team_id, user_id)
);

-- Event results
CREATE TABLE results (
    id SERIAL PRIMARY KEY,
    event_id INTEGER REFERENCES events(id) ON DELETE CASCADE,
    user_id INTEGER REFERENCES users(id),
    team_id INTEGER REFERENCES teams(id),
    round_id INTEGER REFERENCES rounds(id),  -- optional link to full round
    total_strokes INTEGER NOT NULL,
    total_score INTEGER NOT NULL,  -- relative to par
    position INTEGER,
    points_earned INTEGER,
    handicap_used DECIMAL(4,1),
    handicap_score INTEGER,  -- adjusted score
    dnf BOOLEAN DEFAULT FALSE,
    dq BOOLEAN DEFAULT FALSE,
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Prize structure
CREATE TABLE prizes (
    id SERIAL PRIMARY KEY,
    league_id INTEGER REFERENCES leagues(id) ON DELETE CASCADE,
    position INTEGER NOT NULL,
    amount_usd DECIMAL(10,2),
    amount_rgdg INTEGER,  -- P1: token prizes
    UNIQUE(league_id, position)
);

-- Blockchain transactions (P1)
CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    tx_type VARCHAR(30) CHECK (tx_type IN ('purchase', 'league_fee', 'membership', 'prize_payout', 'transfer')),
    amount DECIMAL(20,8) NOT NULL,
    token VARCHAR(10) DEFAULT 'RGDG',
    tx_hash VARCHAR(66),  -- Ethereum transaction hash
    from_address VARCHAR(42),
    to_address VARCHAR(42),
    status VARCHAR(20) DEFAULT 'pending' CHECK (status IN ('pending', 'confirmed', 'failed')),
    event_id INTEGER REFERENCES events(id),
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for performance
CREATE INDEX idx_rounds_user ON rounds(user_id);
CREATE INDEX idx_rounds_layout ON rounds(layout_id);
CREATE INDEX idx_results_event ON results(event_id);
CREATE INDEX idx_results_user ON results(user_id);
CREATE INDEX idx_events_league ON events(league_id);
CREATE INDEX idx_events_date ON events(event_date);
CREATE INDEX idx_transactions_user ON transactions(user_id);
CREATE INDEX idx_transactions_status ON transactions(status);
```

## 9. API Specification

### Authentication Endpoints

```yaml
POST /api/v1/auth/register:
  summary: Register new user
  body:
    email: string (required)
    username: string (required)
    password: string (required)
    display_name: string
  response: { user, access_token, refresh_token }

POST /api/v1/auth/login:
  summary: Login with email/password
  body:
    email: string
    password: string
  response: { user, access_token, refresh_token }

POST /api/v1/auth/web3/nonce:       # P1
  summary: Get nonce for wallet signature
  body:
    wallet_address: string
  response: { nonce, message }

POST /api/v1/auth/web3/verify:      # P1
  summary: Verify signed message
  body:
    wallet_address: string
    signature: string
  response: { user, access_token }

POST /api/v1/auth/refresh:
  summary: Refresh access token
  body:
    refresh_token: string
  response: { access_token }

POST /api/v1/auth/logout:
  summary: Invalidate tokens
```

```yaml
volumes:
  postgres_data:
  redis_data:
```

## Appendix B: Quick Reference

### Points Calculation Formula

```
points = num_participants - finish_position + 1

Example (8 players):
1st place: 8 - 1 + 1 = 8 points
2nd place: 8 - 2 + 1 = 7 points
8th place: 8 - 8 + 1 = 1 point
```

### User Roles

|Role  |Permissions                                         |
|------|----------------------------------------------------|
|Guest |View courses, leaderboards (read-only)              |
|Player|All guest + play rounds, enter leagues              |
|Admin |All player + manage leagues, enter results, treasury|
